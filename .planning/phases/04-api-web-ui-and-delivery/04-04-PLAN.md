Run the following sequence to build the binary and execute automated smoke tests:

```bash
cd /Users/ahmethakanbesel/code-p/pbin

# 1. Full build — must succeed with no CGO
CGO_ENABLED=0 go build -o /tmp/pbin-test ./cmd/pbin
echo "Build exit: $?"

# 2. Start server in background (default config — no auth, default port 8080)
/tmp/pbin-test &
SERVER_PID=$!
sleep 1

# 3. Health check
curl -sf http://localhost:8080/health

# 4. UI pages
curl -sI http://localhost:8080/ | grep "HTTP/"
curl -sI http://localhost:8080/paste | grep "HTTP/"
curl -sI http://localhost:8080/bucket | grep "HTTP/"

# 5. File upload API
UPLOAD=$(curl -s -F "file=@/etc/hostname" -F "expiry=10m" http://localhost:8080/api/upload)
echo "Upload response: $UPLOAD"
FILE_URL=$(echo $UPLOAD | grep -o '"url":"[^"]*"' | cut -d'"' -f4)
echo "File URL: $FILE_URL"

# 6. Download uploaded file
curl -sI "http://localhost:8080/${FILE_URL##*/}" | grep "HTTP/"

# 7. Paste API
PASTE=$(curl -s -X POST -H "Content-Type: application/json" \
  -d '{"content":"hello world","lang":"plaintext","expiry":"10m"}' \
  http://localhost:8080/api/paste)
echo "Paste response: $PASTE"

# 8. Test auth rejection (enable auth via env var, restart)
kill $SERVER_PID
sleep 1
PBIN_AUTH_ENABLED=true PBIN_AUTH_USERNAME=admin PBIN_AUTH_PASSWORD=secret /tmp/pbin-test &
AUTH_PID=$!
sleep 1

AUTH_RESULT=$(curl -s -o /dev/null -w "%{http_code}" -F "file=@/etc/hostname" http://localhost:8080/api/upload)
echo "Auth rejection status (expect 401): $AUTH_RESULT"

AUTHED_RESULT=$(curl -s -o /dev/null -w "%{http_code}" -u admin:secret -F "file=@/etc/hostname" http://localhost:8080/api/upload)
echo "Authed upload status (expect 201): $AUTHED_RESULT"

kill $AUTH_PID
```
