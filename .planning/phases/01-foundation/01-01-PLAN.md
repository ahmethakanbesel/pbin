Wait — koanf does not have a Static provider in the standard packages. Use rawbytes or confmap providers, OR simply set defaults using a plain struct literal approach. The correct pattern with koanf/v2 is:

```go
package config

import (
    "fmt"
    "strings"

    "github.com/knadh/koanf/parsers/toml"
    "github.com/knadh/koanf/providers/env"
    "github.com/knadh/koanf/providers/file"
    "github.com/knadh/koanf/v2"
)

type Config struct {
    Server   ServerConfig
    Database DatabaseConfig
    Storage  StorageConfig
    Auth     AuthConfig
    Upload   UploadConfig
}

type ServerConfig struct {
    Port int
    Host string
}

type DatabaseConfig struct {
    Path string
}

type StorageConfig struct {
    Path string
}

type AuthConfig struct {
    Enabled  bool
    Username string
    Password string
}

type UploadConfig struct {
    MaxBytes int64
}

func defaults() Config {
    return Config{
        Server:   ServerConfig{Port: 8080, Host: "0.0.0.0"},
        Database: DatabaseConfig{Path: "./data/pbin.db"},
        Storage:  StorageConfig{Path: "./data/uploads"},
        Auth:     AuthConfig{Enabled: false},
        Upload:   UploadConfig{MaxBytes: 100 * 1024 * 1024},
    }
}

// Parse loads configuration. configPath may be empty or point to a non-existent file;
// defaults are used in both cases. Environment variables prefixed with PBIN_ override
// any file setting (e.g., PBIN_SERVER_PORT=9090).
func Parse(configPath string) (Config, error) {
    cfg := defaults()

    k := koanf.New(".")

    // Load from TOML file if provided and exists
    if configPath != "" {
        if err := k.Load(file.Provider(configPath), toml.Parser()); err != nil {
            // Non-existent file is acceptable; other errors are not
            if !strings.Contains(err.Error(), "no such file") {
                return cfg, fmt.Errorf("loading config file %s: %w", configPath, err)
            }
        }
    }

    // Load from environment variables with PBIN_ prefix
    // Env key transform: PBIN_SERVER_PORT -> server.port
    if err := k.Load(env.Provider("PBIN_", ".", func(s string) string {
        s = strings.ToLower(strings.TrimPrefix(s, "PBIN_"))
        s = strings.ReplaceAll(s, "_", ".")
        return s
    }), nil); err != nil {
        return cfg, fmt.Errorf("loading env config: %w", err)
    }

    // Unmarshal into struct (only overwrite if keys are present in koanf)
    if k.Exists("server.port") {
        cfg.Server.Port = k.Int("server.port")
    }
    if k.Exists("server.host") {
        cfg.Server.Host = k.String("server.host")
    }
    if k.Exists("database.path") {
        cfg.Database.Path = k.String("database.path")
    }
    if k.Exists("storage.path") {
        cfg.Storage.Path = k.String("storage.path")
    }
    if k.Exists("auth.enabled") {
        cfg.Auth.Enabled = k.Bool("auth.enabled")
    }
    if k.Exists("auth.username") {
        cfg.Auth.Username = k.String("auth.username")
    }
    if k.Exists("auth.password") {
        cfg.Auth.Password = k.String("auth.password")
    }
    if k.Exists("upload.max_bytes") {
        cfg.Upload.MaxBytes = k.Int64("upload.max_bytes")
    }

    return cfg, nil
}
```

```go
package slug

import (
    "crypto/rand"
    "errors"
    "math/big"
)

// charset is the alphabet for generated slugs.
// URL-safe (no +, /, = from base64), unambiguous (no 0/O, 1/l/I).
const charset = "abcdefghijkmnpqrstuvwxyzABCDEFGHJKLMNPQRSTUVWXYZ23456789"

var charsetLen = big.NewInt(int64(len(charset)))

// ErrInvalidLength is returned when n <= 0.
var ErrInvalidLength = errors.New("slug: length must be greater than zero")

// New generates a cryptographically random URL-safe slug of length n.
// Returns ErrInvalidLength if n <= 0.
func New(n int) (string, error) {
    if n <= 0 {
        return "", ErrInvalidLength
    }
    b := make([]byte, n)
    for i := range b {
        idx, err := rand.Int(rand.Reader, charsetLen)
        if err != nil {
            return "", err
        }
        b[i] = charset[idx.Int64()]
    }
    return string(b), nil
}

// MustNew generates a slug of length n and panics on error.
// Intended for use in tests and init code where errors are unrecoverable.
func MustNew(n int) string {
    s, err := New(n)
    if err != nil {
        panic(err)
    }
    return s
}
```
